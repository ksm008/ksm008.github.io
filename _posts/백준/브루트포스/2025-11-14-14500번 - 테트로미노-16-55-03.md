클래스 3의 문제이다. 5가지의 테트로미노를 종이 위에 놓았을 때, 숫자의 합이 가장 클 때를 출력하면 된다. 테트로미노를 대칭, 회전을 시킬 수 있으므로 나올 수 있는 경우의 수는 19가지이다. 이 19가지를 전부 좌표로 하드코딩하여 돌려도 되겠지만 그렇게 하기엔 뭔가 공부에 의미가 없을 것 같았다. 

하드코딩을 하지 않으려면 백트래킹을 사용하는 DFS를 사용해서 구해야 한다. 테트로미노의 형태들은 깊이가 4인 DFS를 돌리는 것과 같다. 딱 하나, T형만 빼고. T형은 DFS로는 구할 수 없는 형태이기 때문에 따로 함수를 만들어 체크해야한다. 

```java
import java.io.*;  
import java.util.StringTokenizer;  
  
public class Main {  
    static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
    static BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));  
    static StringBuilder sb = new StringBuilder();  
  
    static int N, M, maxSum;  
    static boolean[][] visited;  
    static int[][] paper;  
  
    public static void main(String[] args) throws IOException {  
        StringTokenizer st = new StringTokenizer(br.readLine());  
        N = Integer.parseInt(st.nextToken());  
        M = Integer.parseInt(st.nextToken());  
  
        visited = new boolean[N][M];  
  
        paper = new int[N][M];  
  
        for (int i = 0; i < N; i++) {  
            st = new StringTokenizer(br.readLine());  
            for (int j = 0; j < M; j++) {  
                paper[i][j] = Integer.parseInt(st.nextToken());  
            }  
        }  
  
        for (int i = 0; i < N; i++) {  
            for (int j = 0; j < M; j++) {  
                visited[i][j] = true;  
                dfs(i, j, 1, paper[i][j]);  
                visited[i][j] = false;  
                checkT(i, j);  
            }  
        }  
  
        sb.append(maxSum);  
        bw.write(sb.toString());  
        bw.flush();  
        bw.close();  
        br.close();  
    }  
  
    // T를 제외한 나머지 테트로미노는 깊이가 4인 DFS로 전부 구해진다.  
    static void dfs(int c, int r, int depth, int sum) {  
        if (depth == 4) {  
            maxSum = Math.max(maxSum, sum);  
            return;  
        }  
        int[] dy = {-1, 1, 0, 0};  
        int[] dx = {0, 0, -1, 1};  
  
        for (int i = 0; i < 4; i++) {  
            int n = c + dy[i];  
            int m = r + dx[i];  
  
            if (n >= 0 && n < N && m >= 0 && m < M && !visited[n][m]) {  
                visited[n][m] = true;  
                dfs(n, m, depth + 1, sum + paper[n][m]);  
                visited[n][m] = false;  
            }  
        }  
    }  
  
    static void checkT(int c, int r) {  
        int sum;  
        // 아래를 향하는 T        if (r - 1 >= 0 && r + 1 < M && c + 1 < N) {  
            sum = paper[c][r] + paper[c][r - 1] + paper[c][r + 1] + paper[c + 1][r];  
            maxSum = Math.max(maxSum, sum);  
        }  
        // 위를 향하는 T        if (r + 1 < M && c - 1 >= 0 && r - 1 >= 0) {  
            sum = paper[c][r] + paper[c][r + 1] + paper[c][r - 1] + paper[c - 1][r];  
            maxSum = Math.max(maxSum, sum);  
        }  
        // 오른쪽 T        if (r + 1 < M && c + 1 < N && c - 1 >= 0) {  
            sum = paper[c][r] + paper[c + 1][r] + paper[c - 1][r] + paper[c][r + 1];  
            maxSum = Math.max(maxSum, sum);  
        }  
        // 왼쪽 T        if (c + 1 < N && c - 1 >= 0 && r - 1 >= 0) {  
            sum = paper[c][r] + paper[c + 1][r] + paper[c - 1][r] + paper[c][r - 1];  
            maxSum = Math.max(maxSum, sum);  
        }  
    }  
}
```
