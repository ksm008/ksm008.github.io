처음에는 무식한 BFS를 짜봤다. 근데 너무 코드가 지저분하고 돌아가는 느낌이 있다. 그래서 이전에 풀었던 [11403번 - 경로 찾기](https://ksm008.github.io/posts/11403%EB%B2%88-%EA%B2%BD%EB%A1%9C-%EC%B0%BE%EA%B8%B0-17-02-48/) 문제 에서 사용했던 플로이드 워셜 알고리즘을 사용하는 것이 더 나아보였다.

```java
import java.io.*;  
import java.util.*;  
  
public class Main {  
    static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
    static BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));  
    static StringBuilder sb = new StringBuilder();  
  
    static int N;  
  
    static ArrayList<Integer>[] arr;  
    static Deque<int[]> queue;  
    static boolean[] visited;  
  
    public static void main(String[] args) throws IOException {  
        StringTokenizer st = new StringTokenizer(br.readLine());  
  
        N = Integer.parseInt(st.nextToken());  
        int M = Integer.parseInt(st.nextToken());  
  
        queue = new ArrayDeque<>();  
        visited = new boolean[N + 1];  
        arr = new ArrayList[N + 1];  
  
        for (int i = 1; i <= N; i++) {  
            arr[i] = new ArrayList<>();  
        }  
  
        for (int i = 0; i < M; i++) {  
            st = new StringTokenizer(br.readLine());  
            int A = Integer.parseInt(st.nextToken());  
            int B = Integer.parseInt(st.nextToken());  
  
            arr[A].add(B);  
            arr[B].add(A);  
        }  
  
        int result = 1;  
        int lowest = bfs(1);  
  
        for (int i = 2; i <= N; i++) {  
            int curr = bfs(i);  
            if (curr < lowest) {  
                lowest = curr;  
                result = i;  
            }  
        }  
  
        sb.append(result);  
        bw.write(sb.toString());  
        bw.flush();  
        bw.close();  
        br.close();  
    }  
  
    static int bfs(int start) {  
        // 결과 변수  
        int result = 0;  
  
        // 처음엔 1부터 시작. 여기서 i는 종점임.  
        for (int i = 1; i <= N; i++) {  
            // 만약 반복 변수가 시작점이 아닌 경우  
            if (i != start) {  
                // 방문 배열이랑 큐를 초기화 시키기...  
                Arrays.fill(visited, false);  
                queue.clear();  
                // 단계 설정  
                int step = 1;  
                queue.add(new int[]{i, step});  
                // bfs 시작  
                outer:  
                while (!queue.isEmpty()) {  
                    // 큐 안 친구 빼기  
                    int[] curr = queue.poll();  
  
                    // 친구의 숫자와 단계를 저장하고 방문 처리  
                    int curNum = curr[0];  
                    int curStep = curr[1];  
                    visited[curNum] = true;  
  
                    for (Integer next : arr[curNum]) {  
                        if (!visited[next]) {  
                            if (next == start) {  
                                result += curStep;  
                                break outer;  
                            } else {  
                                queue.add(new int[]{next, curStep + 1});  
                                visited[next] = true;  
                            }  
                        }  
                    }  
                }  
            }  
        }  
        return result;  
    }  
}
```
