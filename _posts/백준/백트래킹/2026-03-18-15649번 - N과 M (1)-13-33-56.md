백트래킹 알고리즘을 다루는 기초적인 문제이다. 백트래킹은 재귀를 사용하여 모든 경우의 수를 탐색하지만, 길이 아니다 싶으면 다시 뒤로 돌아가는 방법이다. 알고리즘의 핵심은 아래 2가지이다.
1. 현재 진행하려고 하는 길이 정답이 되는 길인지 판단해야 함
2. 더 이상 갈 수 없는 길이라면 다시 이전 단계로 돌아와야 함

백트래킹 알고리즘은 방문 여부를 기억해야 하므로 불린 배열을 통해 체크한다.

```java
import java.io.*;  
import java.util.StringTokenizer;  
  
public class Main {  
    static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
    static BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));  
    static StringBuilder sb = new StringBuilder();  
  
    static int[] temp;  
    static boolean[] visited;  
  
    public static void main(String[] args) throws IOException {  
        StringTokenizer st = new StringTokenizer(br.readLine());  
        int N = Integer.parseInt(st.nextToken());  
        int M = Integer.parseInt(st.nextToken());  
  
        temp = new int[M];  
        visited = new boolean[N + 1];  
  
        recursive(0, N, M);  
        bw.write(sb.toString());  
        bw.flush();  
        bw.close();  
        br.close();  
    }  
  
    static void recursive(int depth, int N, int M) {  
        if (depth == M) {  
            for (int i = 0; i < M; i++) {  
                if (i != M - 1) {  
                    sb.append(temp[i] + " ");  
                } else {  
                    sb.append(temp[i] + "\n");  
                }  
            }  
            return;  
        }  
  
        for (int i = 1; i <= N; i++) {  
            if (!visited[i]) {  
                visited[i] = true;  
                temp[depth] = i;  
                recursive(depth + 1, N, M);  
                visited[i] = false;  
            }  
        }  
    }  
}
```
