재귀함수 짜는게 너무 헷갈린다. 그냥 내 실력이 딸려서 어떻게 구현해야 할지 감이 잘 안잡힌다.
그나마 알게 된 것이라면... 재귀함수는 특정 트리거를 건드릴 때만 다시 호출한다... 정도?

```java
import java.io.*;  
import java.util.StringTokenizer;  
  
public class Main {  
    static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
    static BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));  
  
    static int num;  
    static int[][] paper;  
    static int whitePaper = 0, bluePaper = 0;  
  
    public static void main(String[] args) throws IOException {  
        num = Integer.parseInt(br.readLine());  
        paper =  new int[num][num];  
  
        for (int i = 0; i < num; i++) {  
            StringTokenizer st = new StringTokenizer(br.readLine());  
            for (int j = 0; j < num; j++) {  
                paper[i][j] = Integer.parseInt(st.nextToken());  
            }  
        }  
  
        recursive(0, 0, num);  
  
        bw.write(whitePaper + "\n" + bluePaper);  
        bw.flush();  
        bw.close();  
        br.close();  
    }  
  
    static void recursive(int row, int col, int size) throws IOException {  
        if (size == 1) {  
            if (paper[row][col] == 0) whitePaper++;  
            else bluePaper++;  
            return;  
        }  
  
        int first = paper[row][col];  
        boolean isSame = true;  
  
        for (int i = row; i < row + size; i++) {  
            for (int j = col; j < col + size; j++) {  
                if (paper[i][j] != first) {  
                    isSame = false;  
                    break;  
                }  
            }  
            if (!isSame) break;  
        }  
  
        if (isSame) {  
            if (first == 0) whitePaper++;  
            else bluePaper++;  
            return;  
        }  
  
        int half = size / 2;  
        recursive(row, col, half);  
        recursive(row, col + half, half);  
        recursive(row + half, col, half);  
        recursive(row + half, col + half, half);  
    }  
}
```
